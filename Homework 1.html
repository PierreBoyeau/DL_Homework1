def normalize_train_test(x_train, x_test):
    mean, std = x_train.mean(), x_train.std()
    x_train = (x_train - mean) / std
    x_test = (x_test - mean) / std
    return x_train, x_test

def normalize_vertices(Y):
    Y_vals = Y.reshape((-1, 3, 2))
    new_y = []
    for positions in Y_vals:
        sorted_vertices  = np.argsort(positions[:, 0])
        sorted_positions = positions[sorted_vertices]
        min_positions = sorted_positions[0, :]
#         sorted_positions[1, :] -= min_positions
#         sorted_positions[2, :] -= min_positions
        new_y.append(sorted_positions)
    return np.array(new_y).reshape(-1, 6)

def noise_layer(ipt):
    amplitudes = np.random.uniform(low=0.5, high=0.8)
    noise = K.random_normal(shape=K.shape(ipt), mean=0.0,  # (BATCH_SIZE, 72, 72, 1)
                            stddev=amplitudes)
    
    return ipt + noise

def denoising_autoencoder():
    mdl = Sequential(layers=[
        Lambda(noise_layer, output_shape=(72, 72, 1), input_shape=(72, 72, 1)),
        Conv2D(filters=4, kernel_size=3, activation='relu', padding='same', input_shape=(72, 72, 1)),
        Dropout(rate=0.3),
        MaxPool2D(pool_size=(2, 2)),
        Conv2D(filters=4, kernel_size=3, activation='relu', padding='same',),
        MaxPool2D(pool_size=(2, 2)),
        Conv2D(filters=8, kernel_size=3, activation='relu', padding='same',),
        MaxPool2D(pool_size=(2, 2)),
        BatchNormalization(),
    
        Conv2D(filters=8, kernel_size=3, activation='relu', padding='same',),
        UpSampling2D(size=(2, 2), interpolation='bilinear'),
        Conv2D(filters=4, kernel_size=3, activation='relu', padding='same',),
        UpSampling2D(size=(2, 2), interpolation='bilinear'),
        Conv2D(filters=4, kernel_size=3, activation='relu', padding='same',),
        UpSampling2D(size=(2, 2), interpolation='bilinear'),
        Conv2D(filters=4, kernel_size=3, activation='relu', padding='same',),
        Conv2D(filters=1, kernel_size=1, activation='linear', padding='same',),
#         Conv2D(filters=1, kernel_size=1, activation='linear', padding='same', use_bias=False),
    ])
    mdl.compile(optimizer=Adam(1e-2), 
                    loss='mse')
    return mdl

mdl = denoising_autoencoder()

def postprocess_X(X_pred_renorm):
    X_treated = X_pred_renorm.copy()
    X_treated[X_treated < 0.5] = 0.0
    X_treated[X_treated > 0.5] = 1.0

In this model, I used a lot of noise into the images. This is motivated by the fact that a denoising binary images with low noise is trivial


In above graph, the upper right image represents the denoised prediction we obtain by simply using a $0.5$ threshold. We can clearly see it is very bad, as there still is noise pretty much everywhere.


Lower figures show respectively the scores of the output of the denoising autoencoder, and the corresponding binary denoised image.
There is no more noise corresponding to isolated points. This is a great improvement over the naive method. That being said, figure no longer is a rectangle, du to very high noise.
As I already said, this is also linked to the fact that there is huge noise in these images (Gaussian noise with $\sigma=0.8$)
